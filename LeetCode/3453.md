package leetcode.editor.cn;//给你一个二维整数数组 squares ，其中 squares[i] = [xi, yi, li] 表示一个与 x 轴平行的正方形的左下角坐标和正方形的边长。
// 
//
// 找到一个最小的 y 坐标，它对应一条水平线，该线需要满足它以上正方形的总面积 等于 该线以下正方形的总面积。 
//
// 答案如果与实际答案的误差在 10⁻⁵ 以内，将视为正确答案。 
//
// 注意：正方形 可能会 重叠。重叠区域应该被 多次计数 。 
//
// 
//
// 示例 1： 
//
// 
// 输入： squares = [[0,0,1],[2,2,1]] 
// 
//
// 输出： 1.00000 
//
// 解释： 
//
// 
//
// 任何在 y = 1 和 y = 2 之间的水平线都会有 1 平方单位的面积在其上方，1 平方单位的面积在其下方。最小的 y 坐标是 1。 
//
// 示例 2： 
//
// 
// 输入： squares = [[0,0,2],[1,1,1]] 
// 
//
// 输出： 1.16667 
//
// 解释： 
//
// 
//
// 面积如下： 
//
// 
// 线下的面积：7/6 * 2 (红色) + 1/6 (蓝色) = 15/6 = 2.5。 
// 线上的面积：5/6 * 2 (红色) + 5/6 (蓝色) = 15/6 = 2.5。 
// 
//
// 由于线以上和线以下的面积相等，输出为 7/6 = 1.16667。 
//
// 
//
// 提示： 
//
// 
// 1 <= squares.length <= 5 * 10⁴ 
// squares[i] = [xi, yi, li] 
// squares[i].length == 3 
// 0 <= xi, yi <= 10⁹ 
// 1 <= li <= 10⁹ 
// 所有正方形的总面积不超过 10¹²。 
// 
//
// Related Topics 数组 二分查找 👍 41 👎 0


import java.util.TreeMap;

class Solution3453 {

    public static void main(String[] args) {
        System.out.println(new Solution3453().separateSquares(new int[][]{{0,0,1},{2,2,1}}));
    }
    public double separateSquares(int[][] squares) {
        // x 坐标无意义，只考虑 y 轴
        int minY = Integer.MAX_VALUE;
        int maxY = 0;
        for(int[] square : squares){
            int y = square[1];
            int l = square[2];
            minY = Math.min(minY, y);
            maxY = Math.max(maxY, y + l);
        }

        // 只需要维持一个差分数组就可以了。
        int[] count = new int[maxY - minY + 1];

        long sum = 0;
        for(int[] square : squares){
            int y = square[1], l = square[2];
            count[y - minY] += l;
            count[y - minY + l] -= l;
            sum += (long) l * l;
        }
        int res = 0;
        long pre = 0;


        for(int i = 1; i < count.length; i ++){
            count[i] = count[i - 1] + count[i];
        }



        //需要考虑一个问题，也就是一段空间内都是0，这样得到的值就会偏大
        for(int i = 0; i < count.length; i ++){
            pre += count[i];
            if(pre << 1 >= sum){
                res = i;
                break;
            }
        }
        //答案在res-1和res之间
        return res  + 1 - (double)( pre << 1 - sum) / count[res] / 2 + minY;


    }

    double sS(int[][] squares){
        long totArea = 0;
        TreeMap<Integer, Long> diff = new TreeMap<>();
        for(int[] sq : squares){
            int y = sq[1];
            long l = sq[2];
            totArea += l * l;
            diff.merge(y, l, Long::sum);
            diff.merge(y + (int) l, -l, Long::sum);
        }
        long area = 0;
        long sumL = 0;
        int preY = 0;
        for(var e : diff.entrySet()){
            int y = e.getKey();
            area += sumL * (y - preY);
            if( area * 2 >= totArea){
                return y - (area * 2 - totArea) / (sumL * 2.0);
            }
            preY = y;
            sumL += e.getValue();
        }
        return -1;
    }
}
