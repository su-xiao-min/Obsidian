class Solution1536 {
    public int minSwaps(int[][] grid) {
//        先直接遍历，创建一个数组，记录遍历的结果，看一下每一行最高可以优先到什么层级，
//        还需要一个全局的计数器，优先让上面的数字小一些
//        第一行要求索引1之后的数字全部为1
        int l = grid.length;
        int[] cnt = new int[l];
//        来一个Map计数，保证提前结束
        Map<Integer, Integer> map = new HashMap<>();
//
        for (int i = 0; i < l; i++) {
            int[] rows = grid[i];
            int cur = 0;
            for (int j = 0; j < l; j++) {
                if (rows[j] != 0) {
//                    保证cur 之后的数字全部为0
                    cur = j;
                }
            }
            cnt[i] = cur;
        }


        int res = 0;


        for (int i = 0; i < l; i++) {
            int j;
            for (j = i; j < l; j++) {
                if (cnt[j] <= i) break;
            }
            if(j == l) return -1;
            for (int k = j; k > i; k--) {
                int temp = cnt[k];
                cnt[k] = cnt[k - 1];
                cnt[k - 1] = temp;
                res ++;
            }

        }
        return res;
    }
