class Solution {
    public String robotWithString(String s) {
                var source = s.toCharArray();
        int len = source.length;
        Queue<Character> stack = new ArrayDeque<>();
        var minFromLast = new char[len];
        minFromLast[len - 1] = source[len - 1];
        for(int i = len - 2; i >= 0; i --){
            minFromLast[i] = (char) Math.min(minFromLast[i + 1], source[i]);
        }
        StringBuilder sb = new StringBuilder(len);
//        因为这里需要从minFromLast当中得到后续的最小值，
        //所以必须要有索引
        for(int i = 0; i < len; i ++){
            if(source[i] == minFromLast[i]){
                sb.append(source[i]);
                continue;
            }
            while(!stack.isEmpty() && stack.peek() <= minFromLast[i]){
                sb.append(stack.poll());
            }
            stack.add(source[i]);
        }
        while(!stack.isEmpty()){
            sb.append(stack.poll());
        }
        return sb.toString();
    }
}
